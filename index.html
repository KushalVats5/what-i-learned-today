with spread operator 
	
		try {
			let impPart = ['keyboard', '2gb data'];
			let whatINeed = ['good sleep', 'laptop', ...impPart, 'repeat'];

			console.log(whatINeed);
		} catch(ourFirend) {
			console.log(ourFirend);
		}
